        function startMainTimer() {
            updateTimerDisplay();
            mainTimerInterval = setInterval(() => {
                if (isGameOver) return clearInterval(mainTimerInterval);
                
                timeRemaining--;
                updateTimerDisplay();

                if (timeRemaining <= 0) {
                    clearInterval(mainTimerInterval);
                    handleTimeout();
                }
            }, 1000);
        }

        function updateTimerDisplay() {
            const m = Math.floor(Math.max(timeRemaining, 0) / 60).toString().padStart(2, '0');
            const s = (Math.max(timeRemaining, 0) % 60).toString().padStart(2, '0');
            timerDisplay.innerText = `${m}:${s}`;
            
            if (timeRemaining <= 60 && timeRemaining > 0) {
                timerDisplay.classList.add('danger');
            } else {
                timerDisplay.classList.remove('danger');
            }
        }

        function handleTimeout() {
            isGameOver = true;
            disableInputs();
            unlockBtn.disabled = true;
            switchScreen('failed');
            document.body.style.backgroundColor = "#020202";
            document.body.style.opacity = "0.7";
        }

            Object.values(screens).forEach(s => s.classList.remove('active'));

            }
        }

                    if (e.key === 'Backspace' && !input.value && index > 0) {
                        inputs[index - 1].focus();
                    } else if (e.key === 'Enter') {
                        handleUnlock();
                    }
                });

                input.addEventListener('paste', (e) => {
                    e.preventDefault();
                    const pasteData = (e.clipboardData || window.clipboardData).getData('text').replace(/[^0-9]/g, '');
                    if (!pasteData) return;
                    
                    for (let i = 0; i < inputs.length; i++) {
                        if (i >= index && i - index < pasteData.length) {
                            inputs[i].value = pasteData[i - index];
                        }
                    }
                    
                    const nextFocus = Math.min(index + pasteData.length, inputs.length - 1);
                    inputs[nextFocus].focus();
                    clearErrorUI();
                });
            });
        }

        function getInputValue() {
            return Array.from(inputs).map(inp => inp.value).join('');
        }

        function disableInputs() {
            inputs.forEach(inp => inp.disabled = true);
        }
        function enableInputs() {
            inputs.forEach(inp => {
                inp.disabled = false;
                inp.value = '';
            });
            if (inputs.length > 0) inputs[0].focus();
        }

        // Validation Logic
        function handleUnlock() {
            if (isLocked || isGameOver) return;
            
            const val = getInputValue();
            if (val.length < 4) {
                showError("Please enter 4 digits.");
                triggerShake();
                return;
            }

            if (val === CONFIG.PASSWORD) {
                handleSuccess();
            } else {
                handleFailure();
            }
        }

        function handleFailure() {
            attemptCount++;
            triggerShake();
            inputs.forEach(inp => inp.classList.add('error'));
            
            if (attemptCount >= CONFIG.MAX_ATTEMPTS) {
                startLockout();
            } else {
                errorMsg.innerText = CONFIG.TEXT.incorrect;
                errorMsg.style.display = "block";
                attemptCountSpan.innerText = attemptCount;
                attemptsMsg.style.display = "block";
                
                setTimeout(() => {
                    inputs.forEach(inp => {
                        inp.value = '';
                        inp.classList.remove('error');
                    });
                    inputs[0].focus();
                }, 800);
            }
        }

        function triggerShake() {
            pinGroup.classList.remove('shake');
            void pinGroup.offsetWidth;
            pinGroup.classList.add('shake');
        }

        function clearErrorUI() {
            errorMsg.style.display = "none";
            attemptsMsg.style.display = "none";
            inputs.forEach(inp => inp.classList.remove('error'));
        }

        showError = (msg) => {
            errorMsg.innerText = msg;
            errorMsg.style.display = "block";
            attemptsMsg.style.display = "none";
        };

        // Lockout
        function startLockout() {
            isLocked = true;
            disableInputs();
            unlockBtn.disabled = true;
            
            errorMsg.style.display = "none";
            attemptsMsg.style.display = "none";
            lockMsg.style.display = "block";
            
            let lockTime = CONFIG.LOCK_TIME;
            
            const updateLockUI = () => {
                lockTimerDisplay.innerText = `00:${lockTime.toString().padStart(2, '0')}`;
            };
            
            updateLockUI();
            
            lockTimerInterval = setInterval(() => {
                lockTime--;
                updateLockUI();
                
                if (lockTime <= 0) {
                    clearInterval(lockTimerInterval);
                    endLockout();
                }
            }, 1000);
        }

        function endLockout() {
            if(isGameOver) return;
            isLocked = false;
            attemptCount = 0;
            enableInputs();
            unlockBtn.disabled = false;
            lockMsg.style.display = "none";
            clearErrorUI();
        }

        const sleep = ms => new Promise(r => setTimeout(r, ms));

        // SUCCESS ANIMATION SEQUENCE
        async function handleSuccess() {
            isGameOver = true;
            disableInputs();
            unlockBtn.style.display = "none";
            clearInterval(mainTimerInterval);
            
            switchScreen('animating');

        // Particle System VFX
        function initParticleSystem() {
            const canvas = document.getElementById('particleCanvas');
            const ctx = canvas.getContext('2d');
            canvas.classList.add('active');

            let width = canvas.width = window.innerWidth;
            let height = canvas.height = window.innerHeight;

            window.addEventListener('resize', () => {
                width = canvas.width = window.innerWidth;
                height = canvas.height = window.innerHeight;
            });

            const particles = [];
            const colors = ['#ee4d2d', '#ffb703', '#00e5ff', '#ffffff', '#ff0055'];

            // Initial Burst (Fountain/Explosion from Center)
            const particleCount = 180;
            for (let i = 0; i < particleCount; i++) {
                const angle = Math.random() * Math.PI * 2;
                const speed = Math.random() * 12 + 2;
                particles.push({
                    x: width / 2,
                    y: height / 2,
                    vx: Math.cos(angle) * speed,
                    vy: Math.sin(angle) * speed - 2, // slightly upwards
                    radius: Math.random() * 4 + 2,
                    color: colors[Math.floor(Math.random() * colors.length)],
                    alpha: 1,
                    decay: Math.random() * 0.015 + 0.005,
                    gravity: 0.15,
                    shape: Math.random() > 0.5 ? 'circle' : 'star'
                });
            }

            // Continuous Floating Ambient Particles
            function spawnAmbientParticle() {
                if (particles.length < 250) {
                    particles.push({
                        x: Math.random() * width,
                        y: height + 10,
                        vx: (Math.random() - 0.5) * 1.5,
                        vy: - (Math.random() * 2 + 1),
                        radius: Math.random() * 3 + 1.5,
                        color: colors[Math.floor(Math.random() * colors.length)],
                        alpha: Math.random() * 0.8 + 0.2,
                        decay: 0.003,
                        gravity: -0.01,
                        shape: 'circle'
                    });
                }
            }

            let ambientTimer = setInterval(spawnAmbientParticle, 100);

            function render() {
                ctx.clearRect(0, 0, width, height);

                for (let i = particles.length - 1; i >= 0; i--) {
                    const p = particles[i];
                    p.x += p.vx;
                    p.y += p.vy;
                    p.vy += p.gravity;
                    p.alpha -= p.decay;

                    if (p.alpha <= 0) {
                        particles.splice(i, 1);
                        continue;
                    }

                    ctx.save();
                    ctx.globalAlpha = Math.max(0, p.alpha);
                    ctx.fillStyle = p.color;
                    ctx.shadowBlur = 10;
                    ctx.shadowColor = p.color;

                    ctx.beginPath();
                    ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.restore();
                }

                requestAnimationFrame(render);
            }

            render();
        }

        function playVictorySound() {
            if (!CONFIG.SUCCESS_SOUND || isMuted) return;
            
            try {
                audioContext = new (window.AudioContext || window.webkitAudioContext)();
                
                const playNote = (freq, startTime, duration, type='sine', vol=0.4) => {
                    const osc = audioContext.createOscillator();
                    const gainNode = audioContext.createGain();
                    
                    osc.type = type;
                    osc.frequency.setValueAtTime(freq, audioContext.currentTime + startTime);
                    
                    gainNode.gain.setValueAtTime(0, audioContext.currentTime + startTime);
                    gainNode.gain.linearRampToValueAtTime(vol, audioContext.currentTime + startTime + 0.08);
                    gainNode.gain.exponentialRampToValueAtTime(0.001, audioContext.currentTime + startTime + duration);
                    
                    osc.connect(gainNode);
                    gainNode.connect(audioContext.destination);
                    
                    osc.start(audioContext.currentTime + startTime);
                    osc.stop(audioContext.currentTime + startTime + duration);
                };

                // Futuristic Triumph Chord Fanfare
                playNote(523.25, 0.0, 0.8, 'sine', 0.3);   // C5
                playNote(659.25, 0.1, 0.8, 'sine', 0.3);   // E5
                playNote(783.99, 0.2, 0.8, 'sine', 0.3);   // G5

                playNote(783.99, 0.8, 1.0, 'triangle', 0.3); // G5
                playNote(987.77, 0.9, 1.0, 'triangle', 0.3); // B5
                playNote(1174.66, 1.0, 1.0, 'triangle', 0.3); // D6

                // Grand Final Chord & Sparkle
                playNote(1046.50, 1.5, 3.5, 'sine', 0.5);  // C6
                playNote(1318.51, 1.6, 3.5, 'sine', 0.4);  // E6
                playNote(1567.98, 1.7, 3.5, 'sine', 0.4);  // G6
                playNote(2093.00, 1.9, 4.0, 'sine', 0.3);  // C7

            } catch(e) {
                console.log("Audio playback blocked/unsupported.");
            }
        }
